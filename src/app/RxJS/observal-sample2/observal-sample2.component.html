  source = Observable.create(observer => {
    for (const n of this.numbers) {
      if (n === 3) {
        observer.error('Something went wrong!!!');
      }
      observer.next(n);
    }
    observer.complete();
  });

  this.source.subscribe(
      value => console.log(`value: ${value}`),
      e => console.log(`error: ${e}`),
      () => console.log(`Completed`)
    );
